Future<void> _generateVideo() async {
  // Bước 1: Lấy văn bản từ TextField và chuẩn hóa
  String text = _textController.text.trim();
  text = _normalizeDescription(text);
  
  // Kiểm tra văn bản không rỗng
  if (text.isEmpty) {
    _showErrorDialog('Vui lòng nhập text để tạo video');
    return;
  }
  
  // Bước 2: Hiển thị trạng thái đang tạo video
  setState(() {
    _isGenerating = true;  // Đặt cờ để disable các nút
  });
  
  // Bước 3: Retry logic (thử lại tối đa 3 lần nếu lỗi)
  const maxRetries = 3;
  int retryCount = 0;
  
  while (retryCount < maxRetries) {
    try {
      // Bước 4: Tạo prompt chi tiết từ văn bản bằng GPT-5
      _showLoadingMessage('Đang tạo prompt từ GPT...');
      final prompt = await _openAIService.generatePrompt(text);
      
      // Bước 5: Tạo video từ prompt bằng Sora 2.0 (5 giây)
      _showLoadingMessage('Đang tạo video từ prompt...');
      final videoUrl = await _videoService.generateVideoFromText(prompt);
      
      // Bước 6: Tự động lưu video vào gallery
      await _galleryService.saveVideo(
        videoUrl: videoUrl,
        title: text,
        sourceType: 'text',
      );
      
      // Bước 7: Chuyển đến màn hình phát video
      Navigator.push(
        context,
        MaterialPageRoute(
          builder: (context) => VideoPlayerScreen(videoUrl: videoUrl),
        ),
      );
      
      break; // Thành công, thoát khỏi vòng lặp retry
      
    } catch (e) {
      // Nếu lỗi, tăng số lần retry và thử lại
      retryCount++;
      if (retryCount >= maxRetries) {
        _showErrorDialog('Lỗi khi tạo video: $e');
        break;
      }
    }
  }
  
  // Bước 8: Tắt trạng thái đang tạo video
  setState(() {
    _isGenerating = false;
  });
}

GIẢI THÍCH:
- async/await: Cho phép chờ đợi các thao tác mất thời gian (gọi API)
- setState(): Cập nhật giao diện khi trạng thái thay đổi
- try-catch: Bắt lỗi và xử lý, tránh app bị crash
- Retry logic: Thử lại nếu lỗi (ví dụ: mạng chậm, API quá tải)

Future<String> generateVideoFromText(String prompt) async {
  // Thêm yêu cầu thoại phải bằng tiếng Việt
  prompt += '\n\nNOTE: The dialogues in the video must be in Vietnamese.';
  
  // Chuẩn bị dữ liệu gửi đến API
  final requestBody = {
    'access_token': accessToken,      // Token xác thực
    'domain': domain,                 // Domain của AIVideoAuto
    'model': 'sora_2_0',             // Sử dụng model Sora 2.0
    'privacy': 'PRIVATE',            // Video riêng tư
    'prompt': prompt,                // Prompt đã tạo từ GPT-5
    'duration': '5',                 // Độ dài video: 5 giây
    'ratio': '16:9',                 // Tỷ lệ khung hình
    'project_id': 'default',         // ID dự án
    'images': jsonEncode([]),        // Không có ảnh (chỉ text)
  };
  
  // Gửi request HTTP POST đến API
  final response = await http.post(
    Uri.parse('$baseUrl/create-video'),  // URL API
    headers: {'Content-Type': 'application/x-www-form-urlencoded'},
    body: requestBody,
  );
  
  // Kiểm tra response
  if (response.statusCode == 200) {
    // Parse dữ liệu JSON từ response
    final data = jsonDecode(utf8.decode(response.bodyBytes));
    
    if (data['videoInfo'] != null) {
      // Lấy videoId từ response
      final videoId = data['videoInfo']['id_base'] ?? data['videoInfo']['id'];
      
      // Polling: Kiểm tra trạng thái video mỗi 10 giây
      // Cho đến khi video hoàn thành
      return await _waitForVideoCompletion(videoId);
    }
  }
  
  // Nếu lỗi, throw exception
  throw Exception('Error creating video');
}

GIẢI THÍCH:
- http.post(): Gửi request HTTP đến server
- jsonDecode(): Chuyển đổi chuỗi JSON thành object Dart
- Polling: Kiểm tra định kỳ xem video đã xong chưa (vì tạo video mất thời gian)
- ?? : Toán tử null-coalescing, nếu bên trái null thì dùng bên phải

Future<String> _waitForVideoCompletion(String videoId) async {
  const maxAttempts = 60;              // Tối đa 60 lần kiểm tra
  const pollInterval = Duration(seconds: 10);  // Mỗi 10 giây kiểm tra 1 lần
  
  for (int attempt = 1; attempt <= maxAttempts; attempt++) {
    // Kiểm tra xem người dùng có hủy không
    if (_isCancelled) {
      throw GenerationCancelledException();
    }
    
    // Đợi 10 giây trước khi kiểm tra lần tiếp theo
    await Future.delayed(pollInterval);
    
    // Kiểm tra trạng thái video
    final videoInfo = await checkVideoStatus(videoId);
    final status = videoInfo['status'] as String;
    
    // Nếu video đã hoàn thành
    if (status == 'MEDIA_GENERATION_STATUS_SUCCESSFUL') {
      // Lấy URL để tải video
      final downloadUrl = videoInfo['download_url'] as String?;
      if (downloadUrl != null && downloadUrl.isNotEmpty) {
        return downloadUrl;  // Trả về URL video
      }
    }
    // Nếu video bị lỗi
    else if (status.contains('FAILED') || status.contains('ERROR')) {
      throw Exception('Video generation failed: $status');
    }
    // Nếu chưa xong, tiếp tục vòng lặp
  }
  
  // Nếu quá 10 phút (60 * 10 giây) vẫn chưa xong → timeout
  throw Exception('Video generation timed out');
}

GIẢI THÍCH:
- Polling: Kỹ thuật kiểm tra định kỳ thay vì chờ callback
- Future.delayed(): Tạm dừng code trong một khoảng thời gian
- for loop: Lặp lại cho đến khi video xong hoặc hết thời gian
- Exception: Ném lỗi để hàm gọi xử lý

ĐOẠN CODE 4: HÀM TẠO PROMPT VỚI GPT-5 (openai_service.dart)

Future<String> generatePrompt(String userText) async {
  // Thay thế [topic] trong template bằng văn bản người dùng nhập
  final userPrompt = _userPromptTemplate.replaceAll('[topic]', userText);
  
  // Chuẩn bị messages để gửi đến GPT-5
  final List<Map<String, dynamic>> messages = [
    {'role': 'user', 'content': userPrompt},
  ];
  
  // Gửi request đến OpenAI API
  final response = await http.post(
    Uri.parse('https://api.openai.com/v1/chat/completions'),
    headers: {
      'Authorization': 'Bearer $_apiKey',  // API key để xác thực
      'Content-Type': 'application/json; charset=utf-8',
    },
    body: json.encode({
      'model': 'gpt-5',        // Sử dụng GPT-5
      'messages': messages,     // Prompt đã chuẩn bị
    }),
  );
  
  // Xử lý response
  if (response.statusCode == 200) {
    // Parse JSON response
    final data = json.decode(utf8.decode(response.bodyBytes));
    
    // Lấy nội dung prompt đã được GPT-5 tạo
    final content = data['choices'][0]['message']['content'] as String;
    
    return content.trim();  // Trả về prompt đã tạo
  } else {
    // Nếu lỗi, throw exception
    throw Exception('Error generating prompt');
  }
}

GIẢI THÍCH:
- Template: Mẫu có sẵn, chỉ cần thay thế phần [topic]
- json.encode(): Chuyển object Dart thành chuỗi JSON
- Authorization header: Gửi API key để xác thực
- choices[0]: Lấy kết quả đầu tiên từ danh sách responses

Future<void> saveVideo({
  required String videoUrl,
  required String title,
  required String sourceType,
}) async {
  try {
    // Bước 1: Tải video từ URL về
    final response = await http.get(Uri.parse(videoUrl));
    
    if (response.statusCode == 200) {
      // Bước 2: Lưu video tạm thời vào thư mục temp
      final directory = await getTemporaryDirectory();
      final file = File('${directory.path}/video_${DateTime.now().millisecondsSinceEpoch}.mp4');
      await file.writeAsBytes(response.bodyBytes);
      
      // Bước 3: Lưu vào gallery của thiết bị
      await Gal.putImageBytes(
        file.readAsBytesSync(),  // Đọc bytes của file video
        album: 'HistoVision',     // Tên album trong gallery
        name: title,              // Tên file video
      );
      
      print('✅ Video saved to gallery: $title');
    }
  } catch (e) {
    print('❌ Error saving video: $e');
    throw Exception('Failed to save video: $e');
  }
}

GIẢI THÍCH:
- http.get(): Tải file từ URL về
- getTemporaryDirectory(): Lấy thư mục tạm thời của app
- writeAsBytes(): Ghi dữ liệu bytes vào file
- Gal.putImageBytes(): Plugin để lưu file vào gallery
- DateTime.now(): Tạo tên file unique bằng timestamp

PHẦN 4: CÁC KHÁI NIỆM QUAN TRỌNG
================================================================================

1. ASYNC/AWAIT:
   - async: Hàm bất đồng bộ, có thể chờ đợi
   - await: Chờ kết quả của hàm async
   - Ví dụ: await http.post() → chờ API trả về

2. SETSTATE():
   - Cập nhật state và rebuild UI
   - Flutter tự động vẽ lại widget khi state thay đổi
   - Ví dụ: _isGenerating = true → nút bị disable

3. TRY-CATCH:
   - try: Thử chạy code, nếu lỗi thì vào catch
   - catch: Xử lý lỗi, không để app crash
   - Ví dụ: try { gọi API } catch { hiển thị lỗi }

4. JSON:
   - Định dạng dữ liệu text
   - Dùng để trao đổi giữa app và server
   - jsonDecode(): Chuyển text → object
   - jsonEncode(): Chuyển object → text

5. HTTP REQUEST:
   - GET: Lấy dữ liệu (ví dụ: tải video)
   - POST: Gửi dữ liệu (ví dụ: tạo video)
   - Headers: Thông tin bổ sung (API key, content-type)
   - Body: Dữ liệu gửi kèm

6. POLLING:
   - Kiểm tra định kỳ thay vì chờ callback
   - Dùng khi server không hỗ trợ webhook
   - Ví dụ: Kiểm tra video mỗi 10 giây
